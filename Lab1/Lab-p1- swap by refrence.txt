#include <iostream>

using namespace std;

void swap(int &a, int &b);

int main()
{
    int a=2 , b=3;
    cout << "a=" << a << endl << "b=" << b << endl;
    swap(a,b);
    cout << "a=" << a << endl << "b=" << b << endl;
    return 0;
}

void swap(int &a, int &b)
{
    int temp = a;
    a = b;
    b = temp;
}
